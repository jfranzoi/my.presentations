	\begin{frame}{Commitment: Stima}
		
		\begin{columns}[T]
		    \begin{column}{.5\textwidth}
		
				\begin{itemize}
					\item Complessità
					\begin{itemize}
						\item Mimina, 50\%
						\item Massima, 90\%
					\end{itemize}
					\item Incertezza
					\begin{itemize}
						\item Buffer ($ \sigma $)
					\end{itemize}
				\end{itemize}

				\begin{itemize}
					\item \textbf{Effort}
					\begin{itemize}
						\item Minima + Buffer

				\end{itemize}
		
		    \end{column}
		    \begin{column}{.5\textwidth}
